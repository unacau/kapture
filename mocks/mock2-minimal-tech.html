    QUICK_START

    # Clone repository
    $ git clone https://github.com/williamkapke/kapture.git
    $ cd kapture/server

    # Install and build
    $ npm install
    $ npm run build
    $ npm start

    # Configure Claude Desktop
    $ echo '{"mcpServers": {"kapture": {...}}}' > config.json
